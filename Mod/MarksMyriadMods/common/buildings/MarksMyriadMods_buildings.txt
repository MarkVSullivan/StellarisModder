building_grand_noble_estates = {
	base_buildtime = @b3_time
	icon = "building_palace_2"
	can_build = no
	category = government
	potential = {
		exists = owner
		planet = {
			or = {
				has_building = building_noble_estates
				owner = { is_ai = no }
			}
		}
		owner = { has_valid_civic = civic_aristocratic_elite }
		NOT = { has_modifier = slave_colony }
	}
	
	allow = {
		has_major_upgraded_capital = yes
	}
	
	destroy_trigger = {
		exists = owner
		NOT = {
			owner = { has_civic = civic_aristocratic_elite }
		}
	}
	
	planet_modifier = {
		job_noble_add = 2
		job_knight_errant_add = 3
		job_serf_add = 2
		planet_housing_add = 7
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = @b2_minerals
		}
		upkeep = {
			energy = 4
		}
	}

	prerequisites = {
		"tech_colonial_centralization"
	}
	
	triggered_desc = {
		text = job_noble_effect_desc
	}
	
	triggered_desc = {
		text = job_serf_effect_desc
	}
	
	triggered_desc = {
		text = job_knight_errant_effect_desc
	}	
}

	position_priority = 100

	category = army

	potential = {

	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = @b1_minerals
		}
		upkeep = {
			energy = @b1_upkeep
		}
	}

	destroy_trigger = {
		exists = owner
		owner = { is_ai = yes }
		AND = {
			OR = {
				num_buildings = { type = building_naval_1 value > 1 }
				has_building = building_naval_2
			}
			free_building_slots = 0
			owner = {
				used_naval_capacity_percent < 0.6
				is_at_war = no # don't destroy while at war
			}
		}
	}

	triggered_planet_modifier = {
	#	potential = {
	#		exists = owner
	#		owner = { is_gestalt = no }
	#	}
		modifier = {
			job_quartermaster_add = 2
		}
	}

	prerequisites = {
		"tech_planetary_defenses"
	}

	upgrades = {
		"building_naval_2"
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = no }
		}
		text = job_quartermaster_effect_desc
	}
}
